class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) { return res; }
        queue<TreeNode*> q;
        
        q.push(root);
        
        while (!q.empty()) {
            int size = q.size();
            vector<int> tmp;
            
            for (int i = 0; i < size; i++) {
                TreeNode *t = q.front();
                q.pop();
                tmp.push_back(t->val);
                if (t->left) { q.push(t->left); }
                if (t->right) { q.push(t->right);}
            }
            res.push_back(tmp);
        }
        return res;
    }
};
